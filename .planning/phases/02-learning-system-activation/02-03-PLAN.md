CRITICAL: Do NOT include the recency component from compute_relevance_score because hybrid_search already handles recency via _recency_weight(). Use ONLY the frequency (access_count) and connection_count dimensions.

       record = records_by_id.get(rid)

           continue

       recency = _recency_weight(ts)
       boosted_score = score * (1.0 + recency_weight * recency)

       # Relevance boost: frequency and connection dimensions only (LEARN-05)
       # Avoids double-counting recency (already handled above)
       if _relevance_score is not None:
           access_count = record.get("access_count", 0)
           # Frequency-only relevance: use access_count with zeroed-out time dimensions
           # compute_relevance_score weights: 0.4 freq + 0.4 recency + 0.2 connections
           # We want only freq (0.4) and connections (0.2) -- pass days_since=0 to neutralize recency
           import math
           freq_factor = math.log1p(max(access_count, 0)) / math.log1p(10)
           # Apply as a mild boost: 0.9 to 1.1 range based on frequency
           boosted_score *= (0.9 + 0.2 * min(freq_factor, 1.0))

       scored_records.append((boosted_score, record))

   Note: We compute the frequency factor directly rather than calling compute_relevance_score() to avoid the recency double-count. This is simpler and more transparent. The math.log1p formula is identical to what compute_relevance_score uses internally.

   ```python
   def _update_tiers(self, records: list[dict]) -> int:
       """Update record tiers based on relevance scoring.

       Returns the number of records whose tier was changed.
       """
       try:
           from jarvis_engine.learning.relevance import compute_relevance_score, classify_tier_by_relevance
       except ImportError:
           return 0

       from datetime import datetime
       from jarvis_engine._compat import UTC

       now = datetime.now(UTC)
       changed = 0

       for record in records:
           access_count = record.get("access_count", 0)
           record_id = record.get("record_id")
           if not record_id:
               continue

           # Compute days since access and creation
           last_accessed_str = record.get("last_accessed", "")
           created_str = record.get("ts", "") or record.get("created_at", "")

           days_since_access = 365.0  # Default: very old
           if last_accessed_str:
               try:
                   la = datetime.fromisoformat(last_accessed_str)
                   if la.tzinfo is None:
                       la = la.replace(tzinfo=UTC)
                   days_since_access = max(0.0, (now - la).total_seconds() / 86400.0)
               except (ValueError, TypeError):
                   pass

           days_since_creation = 365.0
           if created_str:
               try:
                   cr = datetime.fromisoformat(created_str)
                   if cr.tzinfo is None:
                       cr = cr.replace(tzinfo=UTC)
                   days_since_creation = max(0.0, (now - cr).total_seconds() / 86400.0)
               except (ValueError, TypeError):
                   pass

           relevance = compute_relevance_score(access_count, days_since_access, days_since_creation)
           new_tier = classify_tier_by_relevance(relevance, days_since_creation)
           current_tier = record.get("tier", "warm")

           if new_tier != current_tier:
               try:
                   self._engine._db.execute(
                       "UPDATE records SET tier = ? WHERE record_id = ?",
                       (new_tier, record_id),
                   )
                   changed += 1
               except Exception as exc:
                   logger.warning("Failed to update tier for %s: %s", record_id, exc)

       if changed > 0:
           try:
               self._engine._db.commit()
           except Exception as exc:
               logger.warning("Failed to commit tier updates: %s", exc)

       return changed
   ```
